    # 表格字段隐藏显示组件

    导入：

    import FieldScreen from '@/pageCommon/FieldScreen' //列表字段隐藏组件
    1

    使用：

     <!-- 隐藏表格字段按钮 -->
<field-screen :Checkboxlist="cities" @get-Fields="handleFilterFieldsNumber" @on-change="handleCheckedCitiesChange"></field-screen>

声明变量：
cities: ['序号', '操作', '产品信息', '产品状态', '品牌与分类', '供应商信息', '采购价格','采购员', '类别', '创建人', '创建时间'],
checkedCities:[], // 绑定值
checkAll:{}, // 判断列表是否显示对象

事件：

// 处理筛选列表字段显示有多少个
handleFilterFieldsNumber(val){
  this.checkAll = val
   // console.log(val)
},

// 勾选事件
handleCheckedCitiesChange(value) {
  // console.log(value)
  this.cities.forEach((item, index)=>{
  let len = index + 1
   this.$set(this.checkAll, "Field" + len, false)
   this.$forceUpdate();
     if(value.length > 0){
      value.forEach(ele=>{
       if(item == ele){
         this.$set(this.checkAll, "Field" + len, true)
        this.$forceUpdate();
       }
     })
    }else{
     this.$set(this.checkAll, "Field" + len, false)
     this.$forceUpdate();
    }
  })
 },
